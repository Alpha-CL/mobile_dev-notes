``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * 单一条件媒体查询
 * 
 * 
 * @type: all, print, screen, speech
 * @condition: min-width, max-height, orientation: landscape, ...
 */

media [type/condition] {

    ...
}


//- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -//


/**
 * 多条件媒体查询
 * 
 * 
 * @type: all, print, screen, speech
 * @operator: and , not only
 * @condition: min-width, max-height, orientation: landscape, ...
 */

media [type] [operator] [condition] {

    ...
}


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### media type

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * media 类型
 * 
 * 
 ** tty, tv, projection, handheld, braille, embossed, aural等集中类型已在 media_v4 中废弃
 */

all             // 所有设备

print           // 打印机设备

screen          // 彩色的电脑屏幕

speech          // 听觉设备( 针对听力障碍的人士，可将页面内容转换为语音 )


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### media characteristic

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * media 特性
 * 
 * 
 * https://drafts.csswg.org/mediaqueries/
 */

width                           // 宽
    min-width
    max-width

height                          // 高
    min-height
    max-height

orientation                     // 方向

    landscape                   // 横屏: 宽度大于宽度
    portrait                    // 竖屏: 高度大于宽度

aspect-ratio                    // 宽高比

-webkit-device-pixel-ratio      // 像素比( webkit 内核私有属性 )


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```

#### media operator

``` javascript
///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////


/**
 * media 运算符
 * 
 * 
 * 
 */

1) and          // 且: 合并多个媒体类型

2) ,            // 或: 匹配某个媒体查询

3) not          // 非: 取媒体查询的相反结果

4) only         // 仅: 仅在媒体查询匹配成功后应用该样式( 向后兼容旧版本浏览器 )


///////////////////////////////////////////////////////////////////////////////////////////////////////////////////////
```
